cmake_minimum_required(VERSION 3.0.2)
project(localizer)

## Usar C++17
add_compile_options(-std=c++17)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  std_msgs
  geometry_msgs
  nav_msgs
  tf2
  tf2_ros
  tf2_geometry_msgs
  beacon_detector
)

find_package(Eigen3 REQUIRED)

catkin_package(
  INCLUDE_DIRS include
  CATKIN_DEPENDS roscpp std_msgs geometry_msgs nav_msgs tf2 tf2_ros tf2_geometry_msgs beacon_detector
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}
  ${EIGEN3_INCLUDE_DIR}
)

## Execut√°vel principal
add_executable(localizer_node
  src/localizer_node.cpp
  src/main.cpp
)

target_link_libraries(localizer_node
  ${catkin_LIBRARIES}
)

add_dependencies(localizer_node
  ${${PROJECT_NAME}_EXPORTED_TARGETS}
  ${catkin_EXPORTED_TARGETS}
)
